// Non-recursive Quicksort

 qsort(int []  array, int left , int right) {
 

  int i ;

     stack.push(left, right) ;

  while (!stack.empty()) {   // While there are unprocessed subarrays
    // Pop Stack
     (l,r ) <== stack.pop() ;

  
     i  = partition(array,l,r) ;

    // Put new subarrays onto Stack  
    if ((l-i) > THRESHOLD) {   // Left partition
      stack.push( l , i-1);
    } else insertion(array, l, i-1) ;

    if ((r-l) > THRESHOLD) {   // Right partition
      stack.push( i+1 , r);
      
    }else insertion(array, i+1, r) ;
  }
   
}
